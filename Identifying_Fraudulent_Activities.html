In [14]:
def match_IP_to_Country(ipAddress=None):
    condition = (ipAddress >= ipAddresstoCountry['lower_bound_ip_address']) & (ipAddress <= ipAddresstoCountry['upper_bound_ip_address'])
    if condition.any():
        return ipAddresstoCountry['country'][condition].to_string(index=False)
    else:
        return 'unknown'

  1. Time difference between signup and purchase time - To check for unusual activities or bot driven transaction

  3. No. of users associated to a unique ip_address
